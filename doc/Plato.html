    # File lib/foodie/plato.rb, line 40
def carbo   
        grtotal = 0
        sum = 0
        @lista.zip(@listagr).each do |normal, gr|
                grtotal += gr
                cant = gr/1000.0
                sum += normal.carbo*cant
        end
        (sum*100)/grtotal
end

def to_s
        string = @nombre + ",Alimentos: "
        @lista.zip(@listagr).each do |normal, gr|
                string += normal.nombre + " " + gr.to_s + " gr "
        end
        string
end
