def each(&block)
      puntero = @head
      while(puntero!=nil) do
              yield puntero.value  #la lista se recorre alreves
              puntero = puntero.prev
      end
end

def gei
      sum = 0
      @node1 = Node.new(nil,nil,nil)
      @node1 = @head
      until @node1 == @tail do 
              sum += @node1.value.gases
              @node1 = @node1.prev
                              
      end
      sum += @tail.value.gases
end

def insert_head(value)  
        if(@head == nil)
                @node = Node.new(value,nil,nil)
                @head = @node
                @tail = @node
        else
                 @node = Node.new(value,nil,@head)
                 @head.next = @node 
                 @head = @node
        end
end

def insert_tail(value)  
        if(@tail == nil)
                @node = Node.new(value,nil,nil)
                @head = @node
                @tail = @node
        else
                 @node = Node.new(value,@tail,nil)
                 @tail.prev = @node 
                 @tail = @node
        end
end

def pop_head
      if (@head == @tail)
              @head = nil
              @tail = nil
      else
              
              @head = @head.prev
              @head.next = nil
      end
end
